94def install_in(p_venv_base_dir, p_venv_name, p_instructions, p_verbose=True): 

95 '''Execute (install) commands in a virtual environment 

96 

97 Parameters 

98 ---------- 

99 p_venv_base_dir 

100 This is the "root" folder of the virtual environment will be 

101 creates in 

102 p_venv_name 

103 The name of the virtual environment. 

104 p_instructions 

105 Instructions to execute in virtual environment 

106 p_verbose 

107 Give feedback (or not) 

108 Default is True 

109 

110 Returns 

111 ------- 

112 subprocess.CompletedProcess 

113 See https://docs.python.org/3.9/library/subprocess.html#subprocess.CompletedProcess 

114 

115 Examples 

116 -------- 

117 # No proper doctest (<<<) because it is os dependent 

118 beetools.install_in( beetools.get_tmp_dir(), 

119 'new-project', 

120 ['echo Installing in VEnv','pip install wheel','echo Done!']) 

121 + sudo -i 

122 Installing in VEnv 

123 Done! 

124 + exit 

125 True 

126 

127 ''' 

128 switches = [] 

129 script_name = 'install_in' 

130 if tools.get_os() == tools.LINUX: 

131 switches = ['-x'] 

132 script_cmds = ['sudo -i << _EOF_'] 

133 elif tools.get_os() == tools.WINDOWS: 

134 script_cmds = [] 

135 if p_verbose: 

136 script_cmds.append('@ECHO OFF') 

137 else: 

138 script_cmds = [] 

139 if p_verbose: 

140 script_cmds.append('@ECHO OFF') 

141 script_cmds.append('{}\n'.format(activate(p_venv_base_dir, p_venv_name))) 

142 for instr in p_instructions: 

143 script_cmds.append(instr) 

144 if tools.get_os() == tools.LINUX: 

145 script_cmds.append('_EOF_') 

146 script_cmds.append('exit') 

147 ret_code = script.exec_batch_in_session( 

148 script_cmds, p_script_name=script_name, p_verbose=p_verbose, p_switches=switches 

149 ) 

150 return ret_code

153def set_up( 

154 p_venv_base_dir, 

155 p_venv_name, 

156 p_package_list = None, 

157 p_verbose=True 

158) -> bool: 

159 '''Create a virtual environment with some defaults 

160 

161 Parameters 

162 ---------- 

163 p_venv_base_dir 

164 This is the "root" folder of the virtual environment will be 

165 creates in 

166 p_venv_name 

167 The name of the virtual environment. 

168 p_package_list 

169 List of packages to install 

170 

171 Returns 

172 ------- 

173 subprocess.CompletedProcess 

174 See https://docs.python.org/3.9/library/subprocess.html#subprocess.CompletedProcess 

175 

176 Examples 

177 -------- 

178 >>> from beetools import set_up, get_tmp_dir 

179 >>> set_up( get_tmp_dir(),'new-project',['pip','wheel'],p_verbose=False) 

180 True 

181 

182 ''' 

183 switches = [] 

184 script_cmds = [] 

185 if tools.get_os() == tools.WINDOWS: 

186 pip_cmd = 'pip' 

187 else: 

188 pip_cmd = 'pip3' 

189 switches = ['-x'] 

190 script_cmds = ['sudo -i << _EOF_'] 

191 script.exec_cmd( 

192 [ 

193 'python', 

194 '-m', 

195 'venv', 

196 get_dir(p_venv_base_dir, p_venv_name), 

197 ], 

198 p_verbose=p_verbose, 

199 ) 

200 script_name = 'set_up' 

201 script_cmds.append('{}\n'.format(activate(p_venv_base_dir, p_venv_name))) 

202 if not p_package_list: 

203 p_package_list = [] 

204 for package in p_package_list: 

205 if package[0] == 'Web': 

206 script_cmds.append('{} install {}'.format(pip_cmd, package[1])) 

207 elif package[0] == 'Local': 

208 script_cmds.append( 

209 '{} install --find-links {} {}'.format(pip_cmd, package[2], package[1]) 

210 ) 

211 if tools.get_os() == tools.LINUX: 

212 script_cmds.append('_EOF_') 

213 script_cmds.append('exit') 

214 ret_code = script.exec_batch_in_session( 

215 script_cmds, p_script_name=script_name, p_verbose=p_verbose, p_switches=switches 

216 ) 

217 return ret_code
